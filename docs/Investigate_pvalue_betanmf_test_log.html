set up environment

library(readr)
library(NNLM)

data.dir = "../bigdata"

log_transform <- function(X){
  return(log(1+X))
}

exp_transform <- function(logX){
  return(exp(logX) -  1)
}

pvals_cts <- function(X,Lam){
    set.seed(123)
    n = nrow(X)
    p = ncol(X)
    C = matrix(runif(n*p), nrow = n)
    pvals = C * ppois(X-1, Lam) + (1-C) * ppois(X, Lam)
    return(pvals)
}
